Google Sheets — Configuración
Ingresá el ID de tu Google Sheet y el Web App URL del script para lectura/escritura completa. El ID lo encontrás en la URL del spreadsheet: docs.google.com/spreadsheets/d/ESTE_ES_EL_ID/edit
Apps Script para escritura
Para que la app pueda guardar datos en tu spreadsheet, necesitás crear un Google Apps Script.
Pasos: En tu spreadsheet → Extensiones → Apps Script → pegá este código → Implementar → Nueva implementación → Aplicación web → Ejecutar como Yo → Acceso a Cualquiera → Copiar URL.
// Pegar en Google Apps Script const SHEET_ID = "TU_SPREADSHEET_ID"; // ← reemplazá con tu ID function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SHEET_ID); if (data.action === "saveProductos") { const sheet = ss.getSheetByName("Productos") || ss.insertSheet("Productos"); sheet.clearContents(); sheet.appendRow(["id","marca","modelo","color","talles_stock","precio"]); data.productos.forEach(p => { sheet.appendRow([p.id, p.marca, p.modelo, p.color, JSON.stringify(p.stock||{}), p.precio]); }); } if (data.action === "saveVenta") { const sheet = ss.getSheetByName("Ventas") || ss.insertSheet("Ventas"); if (sheet.getLastRow() === 0) { sheet.appendRow(["id","fecha","producto_id","marca","modelo","talle","cantidad","precio_unit","total"]); } const v = data.venta; sheet.appendRow([v.id, v.fecha, v.productoId, v.marca, v.modelo, v.talle, v.cantidad, v.precioUnit, v.total]); } return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false, error: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService.createTextOutput("StockZap OK"); }
